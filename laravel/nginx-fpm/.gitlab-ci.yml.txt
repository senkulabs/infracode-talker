stages:
  - build
  - deploy

composer:
  stage: build
  image: ghcr.io/senkulabs/php:8.3-cli
  cache:
    key: ${CI_COMMIT_REF_SLUG}-composer
    paths:
      - vendor/
  script:
    - composer install --verbose --prefer-dist --no-progress --no-interaction --no-dev --optimize-autoloader
  artifacts:
    expire_in: 1 week
    paths:
      - vendor/
  only:
    - main

bun:
  stage: build
  image: docker.io/oven/bun:latest
  cache:
    key: ${CI_COMMIT_REF_SLUG}-bun
    paths:
      - node_modules/
  script:
    - bun install
    - bun run build
  artifacts:
    expire_in: 1 week
    paths:
      - node_modules/
      - public/build/
  only:
    - main

deploy:
  stage: deploy
  image:
    name: deployphp/deployer:v7
    entrypoint: [""]
  before_script:
    - mkdir -p ~/.ssh
    - eval $(ssh-agent -s)
    - echo "$SSH_KNOWN_HOSTS" > ~/.ssh/known_hosts
    - chmod 644 ~/.ssh/known_hosts
    - echo "$SSH_PRIVATE_KEY" | tr -d '\r' | ssh-add - > /dev/null
  script:
    - dep deploy -vvv
  resource_group: production
  only:
    - main
